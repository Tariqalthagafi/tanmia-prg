function downloadImage(imageId, textId) { var canvas = document.createElement('canvas'); var ctx = canvas.getContext('2d'); var image = document.getElementById(imageId); var text = document.getElementById(textId).textContent; canvas.width = image.width; canvas.height = image.height; ctx.drawImage(image, 0, 0); ctx.font = "24px 'IBM Plex Sans Arabic', sans-serif"; ctx.textAlign = "center"; ctx.textBaseline = "middle"; ctx.shadowColor = "black"; ctx.shadowBlur = 7; var textX = canvas.width / 2; var textY = canvas.height * 0.7; // Customizing text color based on the image if (imageId === 'image1') { ctx.fillStyle = "#16ac7d"; // Match the green color of text on the first image } else { ctx.fillStyle = "white"; // White text for the second image } ctx.fillText(text, textX, textY); var link = document.createElement('a'); link.download = 'tanmia.png'; link.href = canvas.toDataURL('image/png'); link.click(); }
